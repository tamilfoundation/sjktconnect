```python
def _validate_speaker(mention, mp_name: str) -> bool:
    """Check if mp_name appears in the Hansard excerpt.

    Checks full name and each word fragment (for surname matching).
    Returns True if found, False if not verifiable.
    """
    if not mp_name:
        return True  # nothing to verify

    search_text = f"{mention.context_before} {mention.verbatim_quote}".lower()

    # Check full name
    if mp_name.lower() in search_text:
        return True

    # Check individual name fragments (skip short words like a/l, bin, binti)
    fragments = [w for w in mp_name.split() if len(w) > 2 and w.lower() not in {
        "a/l", "a/p", "bin", "binti", "b.", "bt.",
    }]
    for fragment in fragments:
        if fragment.lower() in search_text:
            return True

    return False
```

```python
def _run_brief_quality_loop(brief, mentions):
    """Run evaluate/correct loop on a sitting brief (up to 3 attempts)."""
    from parliament.models import QualityLog, MP
    from parliament.services.corrector import correct_brief
    from schools.models import School
    import markdown

    school_names = list(School.objects.values_list("name", flat=True)[:100])
    mp_names = list(MP.objects.values_list("name", flat=True)[:50])
    source_summaries = "\n".join(
        m.ai_summary for m in mentions if m.ai_summary
    )

    # Apply code fixes first
    cleaned_html = apply_code_fixes(brief.summary_html)
    if cleaned_html != brief.summary_html:
        brief.summary_html = cleaned_html
        brief.save(update_fields=["summary_html", "updated_at"])

    quality_flag = "GREEN"
    attempt = 1
    current_html = brief.summary_html

    while attempt <= 3:
        eval_result = _evaluate_brief(
            brief_html=current_html,
            source_summaries=source_summaries,
            school_names=school_names,
            mp_names=mp_names,
        )

        if eval_result.verdict == "PASS":
            quality_flag = "GREEN"
        elif eval_result.verdict == "REJECT":
            quality_flag = "RED"
        else:
            quality_flag = "AMBER"

        QualityLog.objects.create(
            content_type="brief",
            sitting_brief=brief,
            prompt_version="v1",
            model_used="gemini-2.5-flash",
            attempt_number=attempt,
            verdict=eval_result.verdict,
            tier1_results=eval_result.tier1_results,
            tier2_scores=eval_result.tier2_scores,
            tier3_flags=eval_result.tier3_flags,
            corrections_applied=[],
            quality_flag=quality_flag,
        )

        if eval_result.verdict == "PASS":
            break

        if attempt >= 3:
            logger.warning(
                "Brief %s: circuit breaker after %d attempts — %s flag",
                brief.pk, attempt, quality_flag,
            )
            break

        # Attempt correction
        corrected_html = correct_brief(
            current_html=current_html,
            eval_result=eval_result,
            source_summaries=source_summaries,
        )
        if corrected_html:
            # correct_brief returns markdown — convert to HTML
            current_html = markdown.markdown(corrected_html, extensions=["tables"])
            current_html = apply_code_fixes(current_html)
            brief.summary_html = current_html
            brief.save(update_fields=["summary_html", "updated_at"])
        else:
            break

        attempt += 1

    brief.quality_flag = quality_flag
    brief.is_published = (quality_flag == "GREEN")
    brief.save(update_fields=["quality_flag", "is_published", "updated_at"])
```
